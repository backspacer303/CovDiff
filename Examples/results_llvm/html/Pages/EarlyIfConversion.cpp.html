681 /// feasibility. ----

684 /// ----
685 void SSAIfConv::convertIf(SmallVectorImpl &RemovedBlocks, 0

691 ++NumTrianglesConv; 0
692 else ----
693 ++NumDiamondsConv; 0

695 // Move all instructions into Head, except for the terminators. ----

699 Head->splice(InsertionPoint, TBB, TBB->begin(), TBB->getFirstTerminator()); 0
700 } ----
701 if (FBB != Tail) { 0
702 if (Predicate) 0
703 PredicateBlock(FBB, /*ReversePredicate=*/true); 0
704 Head->splice(InsertionPoint, FBB, FBB->begin(), FBB->getFirstTerminator()); 0
705 } ----

707 bool ExtraPreds = Tail->pred_size() != 2; 0
708 if (ExtraPreds) 0

711 replacePHIInstrs(); 0
712 ----
713 // Fix up the CFG, temporarily leave Head without any successors. ----
714 Head->removeSuccessor(TBB); 0

717 TBB->removeSuccessor(Tail, true); 0
718 if (FBB != Tail) 0
719 FBB->removeSuccessor(Tail, true); 0

722 // It should become a single branch or a fallthrough. ----

727 // through to Tail, and we can join the two blocks. ----

730 TBB->eraseFromParent(); 0
731 } ----
732 if (FBB != Tail) { 0

734 FBB->eraseFromParent(); 0
735 } ----
736 ----

745 RemovedBlocks.push_back(Tail); 0
746 Tail->eraseFromParent(); 0
747 } else { ----
748 // We need a branch to Tail, let code placement work it out later. ----
749 LLVM_DEBUG(dbgs() << "Converting to unconditional branch.\n"); 0
750 SmallVector EmptyCond; 0
751 TII->insertBranch(*Head, Tail, nullptr, EmptyCond, HeadDL); 0
