591 /// Helper method for \see EvaluateSymbolAdd(). ----
592 static void AttemptToFoldSymbolOffsetDifference( 0
593 const MCAssembler *Asm, const MCAsmLayout *Layout, ----
594 const SectionAddrMap *Addrs, bool InSet, const MCSymbolRefExpr *&A, ----
595 const MCSymbolRefExpr *&B, int64_t &Addend) { ----
596 if (!A || !B) 0
597 return; 0
598 ----
599 const MCSymbol &SA = A->getSymbol(); 0
600 const MCSymbol &SB = B->getSymbol(); 0
601 ----
602 if (SA.isUndefined() || SB.isUndefined()) 0
603 return; 0
604 ----
605 if (!Asm->getWriter().isSymbolRefDifferenceFullyResolved(*Asm, A, B, InSet)) 0
606 return; 0
607 ----
608 auto FinalizeFolding = [&]() { 0
609 // Pointers to Thumb symbols need to have their low-bit set to allow ----
610 // for interworking. ----
611 if (Asm->isThumbFunc(&SA)) 0
612 Addend |= 1; 0
613 ----
614 // If symbol is labeled as micromips, we set low-bit to ensure ----
615 // correct offset in .gcc_except_table ----
616 if (Asm->getBackend().isMicroMips(&SA)) 0
617 Addend |= 1; 0
618 ----
619 // Clear the symbol expr pointers to indicate we have folded these ----
620 // operands. ----
621 A = B = nullptr; 0
622 }; 0
623 ----
624 const MCFragment *FA = SA.getFragment(); 0
625 const MCFragment *FB = SB.getFragment(); 0
626 const MCSection &SecA = *FA->getParent(); 0
627 const MCSection &SecB = *FB->getParent(); 0
628 if ((&SecA != &SecB) && !Addrs) 0
629 return; 0
630 ----
631 // When layout is available, we can generally compute the difference using the ----
632 // getSymbolOffset path, which also avoids the possible slow fragment walk. ----
633 // However, linker relaxation may cause incorrect fold of A-B if A and B are ----
634 // separated by a linker-relaxable instruction. If the section contains ----
635 // instructions and InSet is false (not expressions in directive like ----
636 // .size/.fill), disable the fast path. ----
637 if (Layout && (InSet || !SecA.hasInstructions() || 0
638 !Asm->getContext().getTargetTriple().isRISCV())) { 0
639 // If both symbols are in the same fragment, return the difference of their ----
640 // offsets. canGetFragmentOffset(FA) may be false. ----
641 if (FA == FB && !SA.isVariable() && !SB.isVariable()) { 0
642 Addend += SA.getOffset() - SB.getOffset(); 0
643 return FinalizeFolding(); 0
644 } ----
645 // One of the symbol involved is part of a fragment being laid out. Quit now ----
646 // to avoid a self loop. ----
647 if (!Layout->canGetFragmentOffset(FA) || !Layout->canGetFragmentOffset(FB)) 0
648 return; 0
649 ----
650 // Eagerly evaluate when layout is finalized. ----
651 Addend += Layout->getSymbolOffset(A->getSymbol()) - 0
652 Layout->getSymbolOffset(B->getSymbol()); 0
653 if (Addrs && (&SecA != &SecB)) 0
654 Addend += (Addrs->lookup(&SecA) - Addrs->lookup(&SecB)); 0
655 ----
656 FinalizeFolding(); 0
657 } else { ----
658 // When layout is not finalized, our ability to resolve differences between ----
659 // symbols is limited to specific cases where the fragments between two ----
660 // symbols (including the fragments the symbols are defined in) are ----
661 // fixed-size fragments so the difference can be calculated. For example, ----
662 // this is important when the Subtarget is changed and a new MCDataFragment ----
663 // is created in the case of foo: instr; .arch_extension ext; instr .if . - ----
664 // foo. ----
665 if (SA.isVariable() || SB.isVariable() || 0
666 FA->getSubsectionNumber() != FB->getSubsectionNumber()) 0
667 return; 0
668 ----
669 // Try to find a constant displacement from FA to FB, add the displacement ----
670 // between the offset in FA of SA and the offset in FB of SB. ----
671 bool Reverse = false; 0
672 if (FA == FB) { 0
673 Reverse = SA.getOffset() < SB.getOffset(); 0
674 } else if (!isa(FA)) { 0
675 Reverse = std::find_if(std::next(FA->getIterator()), SecA.end(), 0
676 [&](auto &I) { return &I == FB; }) != SecA.end(); 0
677 } ----
678 ----
679 uint64_t SAOffset = SA.getOffset(), SBOffset = SB.getOffset(); 0
680 int64_t Displacement = SA.getOffset() - SB.getOffset(); 0
681 if (Reverse) { 0
682 std::swap(FA, FB); 0
683 std::swap(SAOffset, SBOffset); 0
684 Displacement *= -1; 0
685 } ----
686 ----
687 [[maybe_unused]] bool Found = false; 0
688 // Track whether B is before a relaxable instruction and whether A is after ----
689 // a relaxable instruction. If SA and SB are separated by a linker-relaxable ----
690 // instruction, the difference cannot be resolved as it may be changed by ----
691 // the linker. ----
692 bool BBeforeRelax = false, AAfterRelax = false; 0
693 for (auto FI = FB->getIterator(), FE = SecA.end(); FI != FE; ++FI) { 0
694 auto DF = dyn_cast(FI); 0
695 if (DF && DF->isLinkerRelaxable()) { 0
696 if (&*FI != FB || SBOffset != DF->getContents().size()) 0
697 BBeforeRelax = true; 0
698 if (&*FI != FA || SAOffset == DF->getContents().size()) 0
699 AAfterRelax = true; 0
700 if (BBeforeRelax && AAfterRelax) 0
701 return; 0
702 } ----
703 if (&*FI == FA) { 0
704 Found = true; 0
705 break; 0
706 } ----
707 ----
708 int64_t Num; ----
709 if (DF) { 0
710 Displacement += DF->getContents().size(); 0
711 } else if (auto *FF = dyn_cast(FI); 0
712 FF && FF->getNumValues().evaluateAsAbsolute(Num)) { 0
713 Displacement += Num * FF->getValueSize(); 0
714 } else { ----
715 return; 0
716 } ----
717 } ----
718 // If the previous loop does not find FA, FA must be a dummy fragment not in ----
719 // the fragment list (which means SA is a pending label (see ----
720 // flushPendingLabels)). In either case, we can resolve the difference. ----
721 assert(Found || isa(FA)); 0
722 Addend += Reverse ? -Displacement : Displacement; 0
723 FinalizeFolding(); 0
724 } ----
725 } ----

727 /// Evaluate the result of an add between (conceptually) two MCValues. ----
728 /// ----
729 /// This routine conceptually attempts to construct an MCValue: ----
730 /// Result = (Result_A - Result_B + Result_Cst) ----
731 /// from two MCValue's LHS and RHS where ----
732 /// Result = LHS + RHS ----
733 /// and ----
734 /// Result = (LHS_A - LHS_B + LHS_Cst) + (RHS_A - RHS_B + RHS_Cst). ----
735 /// ----
736 /// This routine attempts to aggressively fold the operands such that the result ----
737 /// is representable in an MCValue, but may not always succeed. ----
738 /// ----
739 /// \returns True on success, false if the result is not representable in an ----
740 /// MCValue. ----
741 ----
742 /// NOTE: It is really important to have both the Asm and Layout arguments. ----
743 /// They might look redundant, but this function can be used before layout ----
744 /// is done (see the object streamer for example) and having the Asm argument ----
745 /// lets us avoid relaxations early. ----
746 static bool ----
747 EvaluateSymbolicAdd(const MCAssembler *Asm, const MCAsmLayout *Layout, 0
748 const SectionAddrMap *Addrs, bool InSet, const MCValue &LHS, ----
749 const MCSymbolRefExpr *RHS_A, const MCSymbolRefExpr *RHS_B, ----
750 int64_t RHS_Cst, MCValue &Res) { ----
751 // FIXME: This routine (and other evaluation parts) are *incredibly* sloppy ----
752 // about dealing with modifiers. This will ultimately bite us, one day. ----
753 const MCSymbolRefExpr *LHS_A = LHS.getSymA(); 0
754 const MCSymbolRefExpr *LHS_B = LHS.getSymB(); 0
755 int64_t LHS_Cst = LHS.getConstant(); 0
756 ----
757 // Fold the result constant immediately. ----
758 int64_t Result_Cst = LHS_Cst + RHS_Cst; 0
759 ----
760 assert((!Layout || Asm) && 0
761 "Must have an assembler object if layout is given!"); ----
762 ----
763 // If we have a layout, we can fold resolved differences. ----
764 if (Asm) { 0
765 // First, fold out any differences which are fully resolved. By ----
766 // reassociating terms in ----
767 // Result = (LHS_A - LHS_B + LHS_Cst) + (RHS_A - RHS_B + RHS_Cst). ----
768 // we have the four possible differences: ----
769 // (LHS_A - LHS_B), ----
770 // (LHS_A - RHS_B), ----
771 // (RHS_A - LHS_B), ----
772 // (RHS_A - RHS_B). ----
773 // Since we are attempting to be as aggressive as possible about folding, we ----
774 // attempt to evaluate each possible alternative. ----
775 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, LHS_A, LHS_B, 0
776 Result_Cst); ----
777 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, LHS_A, RHS_B, 0
778 Result_Cst); ----
779 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, RHS_A, LHS_B, 0
780 Result_Cst); ----
781 AttemptToFoldSymbolOffsetDifference(Asm, Layout, Addrs, InSet, RHS_A, RHS_B, 0
782 Result_Cst); ----
783 } ----
784 ----
785 // We can't represent the addition or subtraction of two symbols. ----
786 if ((LHS_A && RHS_A) || (LHS_B && RHS_B)) 0
787 return false; 0
788 ----
789 // At this point, we have at most one additive symbol and one subtractive ----
790 // symbol -- find them. ----
791 const MCSymbolRefExpr *A = LHS_A ? LHS_A : RHS_A; 0
792 const MCSymbolRefExpr *B = LHS_B ? LHS_B : RHS_B; 0
793 ----
794 Res = MCValue::get(A, B, Result_Cst); 0
795 return true; 0
796 } ----

829 const MCAsmLayout *Layout, ----
830 const MCFixup *Fixup, ----
831 const SectionAddrMap *Addrs, ----
832 bool InSet) const { ----
833 ++stats::MCExprEvaluate; 0
834 ----
835 switch (getKind()) { 0
836 case Target: 0
837 return cast(this)->evaluateAsRelocatableImpl(Res, Layout, 0
838 Fixup); 0
839 ----
840 case Constant: 0
841 Res = MCValue::get(cast(this)->getValue()); 0
842 return true; 0
843 ----
844 case SymbolRef: { 0
845 const MCSymbolRefExpr *SRE = cast(this); 0
846 const MCSymbol &Sym = SRE->getSymbol(); 0
847 const auto Kind = SRE->getKind(); 0
848 ----
849 // Evaluate recursively if this is a variable. ----
850 if (Sym.isVariable() && (Kind == MCSymbolRefExpr::VK_None || Layout) && 0
851 canExpand(Sym, InSet)) { 0
852 bool IsMachO = SRE->hasSubsectionsViaSymbols(); 0
853 if (Sym.getVariableValue()->evaluateAsRelocatableImpl( 0
854 Res, Asm, Layout, Fixup, Addrs, InSet || IsMachO)) { 0
855 if (Kind != MCSymbolRefExpr::VK_None) { 0
856 if (Res.isAbsolute()) { 0
857 Res = MCValue::get(SRE, nullptr, 0); 0
858 return true; 0
859 } ----
860 // If the reference has a variant kind, we can only handle expressions ----
861 // which evaluate exactly to a single unadorned symbol. Attach the ----
862 // original VariantKind to SymA of the result. ----
863 if (Res.getRefKind() != MCSymbolRefExpr::VK_None || !Res.getSymA() || 0
864 Res.getSymB() || Res.getConstant()) 0
865 return false; 0
866 Res = 0
867 MCValue::get(MCSymbolRefExpr::create(&Res.getSymA()->getSymbol(), 0
868 Kind, Asm->getContext()), ----
869 Res.getSymB(), Res.getConstant(), Res.getRefKind()); ----
870 } ----
871 if (!IsMachO) 0
872 return true; 0
873 ----
874 const MCSymbolRefExpr *A = Res.getSymA(); 0
875 const MCSymbolRefExpr *B = Res.getSymB(); 0
876 // FIXME: This is small hack. Given ----
877 // a = b + 4 ----
878 // .long a ----
879 // the OS X assembler will completely drop the 4. We should probably ----
880 // include it in the relocation or produce an error if that is not ----
881 // possible. ----
882 // Allow constant expressions. ----
883 if (!A && !B) 0
884 return true; 0
885 // Allows aliases with zero offset. ----
886 if (Res.getConstant() == 0 && (!A || !B)) 0
887 return true; 0
888 } ----
889 } ----
890 ----
891 Res = MCValue::get(SRE, nullptr, 0); 0
892 return true; 0
893 } ----
894 ----

906 return false; 0

908 break; 0

911 if (Value.getSymA() && !Value.getSymB()) 0
912 return false; 0
913 ----
914 // The cast avoids undefined behavior if the constant is INT64_MIN. ----
915 Res = MCValue::get(Value.getSymB(), Value.getSymA(), 0
916 -(uint64_t)Value.getConstant()); 0
917 break; 0
918 case MCUnaryExpr::Not: 0
919 if (!Value.isAbsolute()) 0
920 return false; 0
921 Res = MCValue::get(~Value.getConstant()); 0
922 break; 0
923 case MCUnaryExpr::Plus: 0
924 Res = Value; 0
925 break; 0
926 } ----
927 ----
928 return true; 0
929 } ----
930 ----
931 case Binary: { 0
932 const MCBinaryExpr *ABE = cast(this); 0
933 MCValue LHSValue, RHSValue; 0
934 ----
935 if (!ABE->getLHS()->evaluateAsRelocatableImpl(LHSValue, Asm, Layout, Fixup, 0
936 Addrs, InSet) || 0
937 !ABE->getRHS()->evaluateAsRelocatableImpl(RHSValue, Asm, Layout, Fixup, 0
938 Addrs, InSet)) { ----
939 // Check if both are Target Expressions, see if we can compare them. ----
940 if (const MCTargetExpr *L = dyn_cast(ABE->getLHS())) { 0
941 const MCTargetExpr *R = cast(ABE->getRHS()); 0

955 ----
