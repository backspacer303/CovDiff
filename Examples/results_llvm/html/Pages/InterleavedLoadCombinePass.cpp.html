108 /// First Order Polynomial on an n-Bit Integer Value ----
109 /// ----
110 /// Polynomial(Value) = Value * B + A + E*2^(n-e) ----
111 /// ----
112 /// A and B are the coefficients. E*2^(n-e) is an error within 'e' most ----
113 /// significant bits. It is introduced if an exact computation cannot be proven ----
114 /// (e.q. division by 2). ----
115 /// ----
116 /// As part of this optimization multiple loads will be combined. It necessary ----
117 /// to prove that loads are within some relative offset to each other. This ----
118 /// class is used to prove relative offsets of values loaded from memory. ----
119 /// ----
120 /// Representing an integer in this form is sound since addition in two's ----
121 /// complement is associative (trivial) and multiplication distributes over the ----
122 /// addition (see Proof(1) in Polynomial::mul). Further, both operations ----
123 /// commute. ----
124 // ----
125 // Example: ----
126 // declare @fn(i64 %IDX, <4 x float>* %PTR) { ----
127 // %Pa1 = add i64 %IDX, 2 ----
128 // %Pa2 = lshr i64 %Pa1, 1 ----
129 // %Pa3 = getelementptr inbounds <4 x float>, <4 x float>* %PTR, i64 %Pa2 ----
130 // %Va = load <4 x float>, <4 x float>* %Pa3 ----
131 // ----
132 // %Pb1 = add i64 %IDX, 4 ----
133 // %Pb2 = lshr i64 %Pb1, 1 ----
134 // %Pb3 = getelementptr inbounds <4 x float>, <4 x float>* %PTR, i64 %Pb2 ----
135 // %Vb = load <4 x float>, <4 x float>* %Pb3 ----
136 // ... } ----
137 // ----
138 // The goal is to prove that two loads load consecutive addresses. ----
139 // ----
140 // In this case the polynomials are constructed by the following ----
141 // steps. ----
142 // ----
143 // The number tag #e specifies the error bits. ----
144 // ----
145 // Pa_0 = %IDX #0 ----
146 // Pa_1 = %IDX + 2 #0 | add 2 ----
147 // Pa_2 = %IDX/2 + 1 #1 | lshr 1 ----
148 // Pa_3 = %IDX/2 + 1 #1 | GEP, step signext to i64 ----
149 // Pa_4 = (%IDX/2)*16 + 16 #0 | GEP, multiply index by sizeof(4) for floats ----
150 // Pa_5 = (%IDX/2)*16 + 16 #0 | GEP, add offset of leading components ----
151 // ----
152 // Pb_0 = %IDX #0 ----
153 // Pb_1 = %IDX + 4 #0 | add 2 ----
154 // Pb_2 = %IDX/2 + 2 #1 | lshr 1 ----
155 // Pb_3 = %IDX/2 + 2 #1 | GEP, step signext to i64 ----
156 // Pb_4 = (%IDX/2)*16 + 32 #0 | GEP, multiply index by sizeof(4) for floats ----
157 // Pb_5 = (%IDX/2)*16 + 16 #0 | GEP, add offset of leading components ----

159 // Pb_5 - Pa_5 = 16 #0 | subtract to get the offset ----

775 /// ShuffleVectorInst specialization to compute vector information. ----
776 /// ----
777 /// \param SVI ShuffleVectorInst to operate on ----
778 /// \param Result Result of the computation ----
779 /// ----
780 /// Compute the left and the right side vector information and merge them by ----

782 /// and right side have compatible loads. This means that all loads are with ----
783 /// in the same basic block and are based on the same pointer. ----

799 RHS.BB = nullptr; 0

801 // Neither operand produced sensible results? ----
802 if (!LHS.BB && !RHS.BB) 0
803 return false; 0

807 Result.PV = RHS.PV; 0
808 } ----
809 // Only LHS produced sensible results? ----

812 Result.PV = LHS.PV; 0

814 // Both operands produced sensible results? ----

816 Result.BB = LHS.BB; 0

820 else { ----

827 Result.Is.insert(LHS.Is.begin(), LHS.Is.end()); 0
828 } ----

830 Result.LIs.insert(RHS.LIs.begin(), RHS.LIs.end()); 0
831 Result.Is.insert(RHS.Is.begin(), RHS.Is.end()); 0
832 } ----
833 Result.Is.insert(SVI); 0
834 Result.SVI = SVI; 0
835 ----
836 int j = 0; 0
837 for (int i : SVI->getShuffleMask()) { 0
838 assert((i < 2 * (signed)ArgTy->getNumElements()) && 0

840 ----
841 if (i < 0) 0

855 } ----
856 ----

858 } 0

951 /// ----
952 /// \param Ptr input pointer value ----
953 /// \param Result result polynomial ----
954 /// \param BasePtr pointer the polynomial is based on ----
955 /// \param DL Datalayout of the target machine ----
956 static void computePolynomialFromPointer(Value &Ptr, Polynomial &Result, 0

958 const DataLayout &DL) { ----

960 PointerType *PtrTy = dyn_cast(Ptr.getType()); 0
961 if (!PtrTy) { 0
962 Result = Polynomial(); 0

964 return; 0

975 break; 0

979 break; 0
980 } ----
981 } ----
982 /// Resolve GetElementPtrInst. ----
983 else if (isa(&Ptr)) { 0

985 ----

987 ----
988 // Check if we can compute the Offset with accumulateConstantOffset ----
989 if (GEP.accumulateConstantOffset(DL, BaseOffset)) { 0

1004 } ----
1005 ----

1007 if (idxOperand + 1 != e) { 0

1009 BasePtr = nullptr; 0
1010 return; 0
1011 } ----
1012 ----

1014 computePolynomial(*GEP.getOperand(idxOperand), Result); 0
1015 ----
1016 // Compute base offset from zero based index, excluding the last ----
1017 // variable operand. ----

1020 ----
1021 // Apply the operations of GEP to the polynomial. ----
1022 unsigned ResultSize = DL.getTypeAllocSize(GEP.getResultElementType()); 0

1034 } ----
1035 } ----

1114 bool InterleavedLoadCombineImpl::combine(std::list &InterleavedLoad, 0
1115 OptimizationRemarkEmitter &ORE) { ----
1116 LLVM_DEBUG(dbgs() << "Checking interleaved load\n"); 0
1117 ----
1118 // The insertion point is the LoadInst which loads the first values. The ----
1119 // following tests are used to proof that the combined load can be inserted ----
1120 // just before InsertionPoint. ----
1121 LoadInst *InsertionPoint = InterleavedLoad.front().EI[0].LI; 0
1122 ----
1123 // Test if the offset is computed ----
1124 if (!InsertionPoint) 0
1125 return false; 0
1126 ----
1127 std::set LIs; 0
1128 std::set Is; 0
1129 std::set SVIs; 0
1130 ----
1131 InstructionCost InterleavedCost; 0
1132 InstructionCost InstructionCost = 0; 0
1133 const TTI::TargetCostKind CostKind = TTI::TCK_SizeAndLatency; 0
1134 ----
1135 // Get the interleave factor ----
1136 unsigned Factor = InterleavedLoad.size(); 0
1137 ----
1138 // Merge all input sets used in analysis ----
1139 for (auto &VI : InterleavedLoad) { 0
1140 // Generate a set of all load instructions to be combined ----
1141 LIs.insert(VI.LIs.begin(), VI.LIs.end()); 0
1142 ----
1143 // Generate a set of all instructions taking part in load ----
1144 // interleaved. This list excludes the instructions necessary for the ----
1145 // polynomial construction. ----
1146 Is.insert(VI.Is.begin(), VI.Is.end()); 0
1147 ----
1148 // Generate the set of the final ShuffleVectorInst. ----
1149 SVIs.insert(VI.SVI); 0
1150 } ----
1151 ----
1152 // There is nothing to combine. ----
1153 if (LIs.size() < 2) 0
1154 return false; 0
1155 ----
1156 // Test if all participating instruction will be dead after the ----
1157 // transformation. If intermediate results are used, no performance gain can ----
1158 // be expected. Also sum the cost of the Instructions beeing left dead. ----
1159 for (const auto &I : Is) { 0
1160 // Compute the old cost ----
1161 InstructionCost += TTI.getInstructionCost(I, CostKind); 0
1162 ----
1163 // The final SVIs are allowed not to be dead, all uses will be replaced ----
1164 if (SVIs.find(I) != SVIs.end()) 0
1165 continue; 0
1166 ----
1167 // If there are users outside the set to be eliminated, we abort the ----
1168 // transformation. No gain can be expected. ----
1169 for (auto *U : I->users()) { 0
1170 if (Is.find(dyn_cast(U)) == Is.end()) 0
1171 return false; 0
1172 } ----
1173 } ----
1174 ----
1175 // We need to have a valid cost in order to proceed. ----
1176 if (!InstructionCost.isValid()) 0
1177 return false; 0
1178 ----
1179 // We know that all LoadInst are within the same BB. This guarantees that ----
1180 // either everything or nothing is loaded. ----
1181 LoadInst *First = findFirstLoad(LIs); 0
1182 ----
1183 // To be safe that the loads can be combined, iterate over all loads and test ----
1184 // that the corresponding defining access dominates first LI. This guarantees ----
1185 // that there are no aliasing stores in between the loads. ----
1186 auto FMA = MSSA.getMemoryAccess(First); 0
1187 for (auto *LI : LIs) { 0
1188 auto MADef = MSSA.getMemoryAccess(LI)->getDefiningAccess(); 0
1189 if (!MSSA.dominates(MADef, FMA)) 0
1190 return false; 0
1191 } ----
1192 assert(!LIs.empty() && "There are no LoadInst to combine"); 0
1193 ----
1194 // It is necessary that insertion point dominates all final ShuffleVectorInst. ----
1195 for (auto &VI : InterleavedLoad) { 0
1196 if (!DT.dominates(InsertionPoint, VI.SVI)) 0
1197 return false; 0
1198 } ----
1199 ----
1200 // All checks are done. Add instructions detectable by InterleavedAccessPass ----
1201 // The old instruction will are left dead. ----
1202 IRBuilder<> Builder(InsertionPoint); 0
1203 Type *ETy = InterleavedLoad.front().SVI->getType()->getElementType(); 0
1204 unsigned ElementsPerSVI = ----
1205 cast(InterleavedLoad.front().SVI->getType()) 0
1206 ->getNumElements(); 0
1207 FixedVectorType *ILTy = FixedVectorType::get(ETy, Factor * ElementsPerSVI); 0
1208 ----
1209 auto Indices = llvm::to_vector<4>(llvm::seq(0, Factor)); 0
1210 InterleavedCost = TTI.getInterleavedMemoryOpCost( 0
1211 Instruction::Load, ILTy, Factor, Indices, InsertionPoint->getAlign(), ----
1212 InsertionPoint->getPointerAddressSpace(), CostKind); ----
1213 ----
1214 if (InterleavedCost >= InstructionCost) { 0
1215 return false; 0
1216 } ----
1217 ----
1218 // Create a pointer cast for the wide load. ----
1219 auto CI = Builder.CreatePointerCast(InsertionPoint->getOperand(0), 0
1220 ILTy->getPointerTo(), 0
1221 "interleaved.wide.ptrcast"); ----
1222 ----
1223 // Create the wide load and update the MemorySSA. ----
1224 auto LI = Builder.CreateAlignedLoad(ILTy, CI, InsertionPoint->getAlign(), 0
1225 "interleaved.wide.load"); 0
1226 auto MSSAU = MemorySSAUpdater(&MSSA); 0
1227 MemoryUse *MSSALoad = cast(MSSAU.createMemoryAccessBefore( 0
1228 LI, nullptr, MSSA.getMemoryAccess(InsertionPoint))); 0
1229 MSSAU.insertUse(MSSALoad, /*RenameUses=*/ true); 0
1230 ----
1231 // Create the final SVIs and replace all uses. ----
1232 int i = 0; 0
1233 for (auto &VI : InterleavedLoad) { 0
1234 SmallVector Mask; 0
1235 for (unsigned j = 0; j < ElementsPerSVI; j++) 0
1236 Mask.push_back(i + j * Factor); 0
1237 ----
1238 Builder.SetInsertPoint(VI.SVI); 0
1239 auto SVI = Builder.CreateShuffleVector(LI, Mask, "interleaved.shuffle"); 0
1240 VI.SVI->replaceAllUsesWith(SVI); 0
1241 i++; 0
1242 } 0
1243 ----
1244 NumInterleavedLoadCombine++; 0
1245 ORE.emit([&]() { 0
1246 return OptimizationRemark(DEBUG_TYPE, "Combined Interleaved Load", LI) 0
1247 << "Load interleaved combined with factor " 0
1248 << ore::NV("Factor", Factor); 0
1249 }); ----
1250 ----
1251 return true; 0
1252 } 0

1256 bool changed = false; 0

1258 ----

1260 ----
1261 // Start with the highest factor to avoid combining and recombining. ----
1262 for (unsigned Factor = MaxFactor; Factor >= 2; Factor--) { 0
