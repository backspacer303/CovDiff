521 bool X86MCInstrAnalysis::clearsSuperRegisters(const MCRegisterInfo &MRI, 0
522 const MCInst &Inst, ----
523 APInt &Mask) const { ----
524 const MCInstrDesc &Desc = Info->get(Inst.getOpcode()); 0
525 unsigned NumDefs = Desc.getNumDefs(); 0
526 unsigned NumImplicitDefs = Desc.implicit_defs().size(); 0
527 assert(Mask.getBitWidth() == NumDefs + NumImplicitDefs && 0
528 "Unexpected number of bits in the mask!"); ----
529 ----
530 bool HasVEX = (Desc.TSFlags & X86II::EncodingMask) == X86II::VEX; 0
531 bool HasEVEX = (Desc.TSFlags & X86II::EncodingMask) == X86II::EVEX; 0
532 bool HasXOP = (Desc.TSFlags & X86II::EncodingMask) == X86II::XOP; 0
533 ----
534 const MCRegisterClass &GR32RC = MRI.getRegClass(X86::GR32RegClassID); 0
535 const MCRegisterClass &VR128XRC = MRI.getRegClass(X86::VR128XRegClassID); 0
536 const MCRegisterClass &VR256XRC = MRI.getRegClass(X86::VR256XRegClassID); 0
537 ----
538 auto ClearsSuperReg = [=](unsigned RegID) { 0
539 // On X86-64, a general purpose integer register is viewed as a 64-bit ----
540 // register internal to the processor. ----
541 // An update to the lower 32 bits of a 64 bit integer register is ----
542 // architecturally defined to zero extend the upper 32 bits. ----
543 if (GR32RC.contains(RegID)) 0
544 return true; 0
545 ----
546 // Early exit if this instruction has no vex/evex/xop prefix. ----
547 if (!HasEVEX && !HasVEX && !HasXOP) 0
548 return false; 0
549 ----
550 // All VEX and EVEX encoded instructions are defined to zero the high bits ----
551 // of the destination register up to VLMAX (i.e. the maximum vector register ----
552 // width pertaining to the instruction). ----
553 // We assume the same behavior for XOP instructions too. ----
554 return VR128XRC.contains(RegID) || VR256XRC.contains(RegID); 0
555 }; 0
556 ----
557 Mask.clearAllBits(); 0
558 for (unsigned I = 0, E = NumDefs; I < E; ++I) { 0
559 const MCOperand &Op = Inst.getOperand(I); 0
560 if (ClearsSuperReg(Op.getReg())) 0
561 Mask.setBit(I); 0
562 } ----
563 ----
564 for (unsigned I = 0, E = NumImplicitDefs; I < E; ++I) { 0
565 const MCPhysReg Reg = Desc.implicit_defs()[I]; 0
566 if (ClearsSuperReg(Reg)) 0
567 Mask.setBit(NumDefs + I); 0
568 } ----
569 ----
570 return Mask.getBoolValue(); 0
571 } ----
