708 ----
709 Function *F = CI.getCalledFunction(); 0

711 return false; 0

714 if (ID == Intrinsic::not_intrinsic || !isTriviallyScalariable(ID)) 0

717 // unsigned NumElems = VT->getNumElements(); ----
718 unsigned NumArgs = CI.arg_size(); 0

721 SmallVector Scattered(NumArgs); 0
722 SmallVector OverloadIdx(NumArgs, -1); 0

725 // Add return type if intrinsic is overloaded on it. ----
726 if (isVectorIntrinsicWithOverloadTypeAtArg(ID, -1)) 0
727 Tys.push_back(VS->SplitTy); 0
728 ----
729 // Assumes that any vector type has the same number of elements as the return ----
730 // vector type, which is true for all current intrinsics. ----
731 for (unsigned I = 0; I != NumArgs; ++I) { 0
732 Value *OpI = CI.getOperand(I); 0
733 if (auto *OpVecTy = dyn_cast(OpI->getType())) { 0

738 // happen if the vector elements are different and the ScalarizeMinBits ----
739 // option is used. ----
740 // ----
741 // We could in principle handle this case as well, at the cost of ----
742 // complicating the scattering machinery to support multiple scattering ----
743 // granularities for a single value. ----
744 return false; 0
745 } ----
746 ----
747 Scattered[I] = scatter(&CI, OpI, *OpVS); 0
748 if (isVectorIntrinsicWithOverloadTypeAtArg(ID, I)) { 0

750 Tys.push_back(OpVS->SplitTy); 0
751 } ----
752 } else { ----
753 ScalarOperands[I] = OpI; 0

789 ----
790 gather(&CI, Res, *VS); 0

926 std::optional DstVS = getVectorSplit(BCI.getDestTy()); 0

929 return false; 0
930 ----
931 const bool isPointerTy = DstVS->VecTy->getElementType()->isPointerTy(); 0
932 ----
933 // Vectors of pointers are always fully scalarized. ----
934 assert(!isPointerTy || (DstVS->NumPacked == 1 && SrcVS->NumPacked == 1)); 0

937 Scatterer Op0 = scatter(&BCI, BCI.getOperand(0), *SrcVS); 0

941 unsigned DstSplitBits = DstVS->SplitTy->getPrimitiveSizeInBits(); 0

946 for (unsigned I = 0; I < DstVS->NumFragments; ++I) { 0
947 Res[I] = Builder.CreateBitCast(Op0[I], DstVS->getFragmentType(I), 0
948 BCI.getName() + ".i" + Twine(I)); 0
949 } ----
950 } else if (SrcSplitBits % DstSplitBits == 0) { 0
951 // Convert each source fragment to the same-sized destination vector and ----
952 // then scatter the result to the destination. ----
953 VectorSplit MidVS; 0
954 MidVS.NumPacked = DstVS->NumPacked; 0
955 MidVS.NumFragments = SrcSplitBits / DstSplitBits; 0
956 MidVS.VecTy = FixedVectorType::get(DstVS->VecTy->getElementType(), 0
957 MidVS.NumPacked * MidVS.NumFragments); 0

959 ----
960 unsigned ResI = 0; 0
961 for (unsigned I = 0; I < SrcVS->NumFragments; ++I) { 0

974 for (unsigned J = 0; J < MidVS.NumFragments; ++J) 0
975 Res[ResI++] = Mid[J]; 0
976 } 0
977 } else if (DstSplitBits % SrcSplitBits == 0) { 0
978 // Gather enough source fragments to make up a destination fragment and ----
979 // then convert to the destination type. ----
980 VectorSplit MidVS; 0
981 MidVS.NumFragments = DstSplitBits / SrcSplitBits; 0
982 MidVS.NumPacked = SrcVS->NumPacked; 0
983 MidVS.VecTy = FixedVectorType::get(SrcVS->VecTy->getElementType(), 0
984 MidVS.NumPacked * MidVS.NumFragments); 0
985 MidVS.SplitTy = SrcVS->SplitTy; 0

988 SmallVector ConcatOps; 0
989 ConcatOps.resize(MidVS.NumFragments); 0
990 for (unsigned I = 0; I < DstVS->NumFragments; ++I) { 0
991 for (unsigned J = 0; J < MidVS.NumFragments; ++J) 0
992 ConcatOps[J] = Op0[SrcI++]; 0
993 Value *V = concatenate(Builder, ConcatOps, MidVS, 0
994 BCI.getName() + ".i" + Twine(I)); 0
995 Res[I] = Builder.CreateBitCast(V, DstVS->getFragmentType(I), 0
996 BCI.getName() + ".i" + Twine(I)); 0
997 } ----
998 } else { 0
999 return false; 0
1000 } ----
1001 ----
1002 gather(&BCI, Res, *DstVS); 0
1003 return true; 0
1004 } 0
