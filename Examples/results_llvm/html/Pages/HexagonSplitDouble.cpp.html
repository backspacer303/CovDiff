772 void HexagonSplitDoubleRegs::splitShift(MachineInstr *MI, 0
773 const UUPairMap &PairMap) { ----
774 using namespace Hexagon; ----
775 ----
776 MachineOperand &Op0 = MI->getOperand(0); 0
777 MachineOperand &Op1 = MI->getOperand(1); 0
778 MachineOperand &Op2 = MI->getOperand(2); 0
779 assert(Op0.isReg() && Op1.isReg() && Op2.isImm()); 0
780 int64_t Sh64 = Op2.getImm(); 0
781 assert(Sh64 >= 0 && Sh64 < 64); 0
782 unsigned S = Sh64; 0
783 ----
784 UUPairMap::const_iterator F = PairMap.find(Op0.getReg()); 0
785 assert(F != PairMap.end()); 0
786 const UUPair &P = F->second; 0
787 Register LoR = P.first; 0
788 Register HiR = P.second; 0
789 ----
790 unsigned Opc = MI->getOpcode(); 0
791 bool Right = (Opc == S2_lsr_i_p || Opc == S2_asr_i_p); 0
792 bool Left = !Right; 0
793 bool Signed = (Opc == S2_asr_i_p); 0
794 ----
795 MachineBasicBlock &B = *MI->getParent(); 0
796 DebugLoc DL = MI->getDebugLoc(); 0
797 unsigned RS = getRegState(Op1); 0
798 unsigned ShiftOpc = Left ? S2_asl_i_r 0
799 : (Signed ? S2_asr_i_r : S2_lsr_i_r); ----
800 unsigned LoSR = isub_lo; 0
801 unsigned HiSR = isub_hi; 0
802 ----
803 if (S == 0) { 0
804 // No shift, subregister copy. ----
805 BuildMI(B, MI, DL, TII->get(TargetOpcode::COPY), LoR) 0
806 .addReg(Op1.getReg(), RS & ~RegState::Kill, LoSR); 0
807 BuildMI(B, MI, DL, TII->get(TargetOpcode::COPY), HiR) 0
808 .addReg(Op1.getReg(), RS, HiSR); 0
809 } else if (S < 32) { 0
810 const TargetRegisterClass *IntRC = &IntRegsRegClass; 0
811 Register TmpR = MRI->createVirtualRegister(IntRC); 0
812 // Expansion: ----
813 // Shift left: DR = shl R, #s ----
814 // LoR = shl R.lo, #s ----
815 // TmpR = extractu R.lo, #s, #32-s ----
816 // HiR = or (TmpR, asl(R.hi, #s)) ----
817 // Shift right: DR = shr R, #s ----
818 // HiR = shr R.hi, #s ----
819 // TmpR = shr R.lo, #s ----
820 // LoR = insert TmpR, R.hi, #s, #32-s ----
821 ----
822 // Shift left: ----
823 // LoR = shl R.lo, #s ----
824 // Shift right: ----
825 // TmpR = shr R.lo, #s ----
826 ----
827 // Make a special case for A2_aslh and A2_asrh (they are predicable as ----
828 // opposed to S2_asl_i_r/S2_asr_i_r). ----
829 if (S == 16 && Left) 0
830 BuildMI(B, MI, DL, TII->get(A2_aslh), LoR) 0
831 .addReg(Op1.getReg(), RS & ~RegState::Kill, LoSR); 0
832 else if (S == 16 && Signed) 0
833 BuildMI(B, MI, DL, TII->get(A2_asrh), TmpR) 0
834 .addReg(Op1.getReg(), RS & ~RegState::Kill, LoSR); 0
835 else ----
836 BuildMI(B, MI, DL, TII->get(ShiftOpc), (Left ? LoR : TmpR)) 0
837 .addReg(Op1.getReg(), RS & ~RegState::Kill, LoSR) 0
838 .addImm(S); 0
839 ----
840 if (Left) { 0
841 // TmpR = extractu R.lo, #s, #32-s ----
842 BuildMI(B, MI, DL, TII->get(S2_extractu), TmpR) 0
843 .addReg(Op1.getReg(), RS & ~RegState::Kill, LoSR) 0
844 .addImm(S) 0
845 .addImm(32-S); 0
846 // HiR = or (TmpR, asl(R.hi, #s)) ----
847 BuildMI(B, MI, DL, TII->get(S2_asl_i_r_or), HiR) 0
848 .addReg(TmpR) 0
849 .addReg(Op1.getReg(), RS, HiSR) 0
850 .addImm(S); 0
851 } else { ----
852 // HiR = shr R.hi, #s ----
853 BuildMI(B, MI, DL, TII->get(ShiftOpc), HiR) 0
854 .addReg(Op1.getReg(), RS & ~RegState::Kill, HiSR) 0
855 .addImm(S); 0
856 // LoR = insert TmpR, R.hi, #s, #32-s ----
857 BuildMI(B, MI, DL, TII->get(S2_insert), LoR) 0
858 .addReg(TmpR) 0
859 .addReg(Op1.getReg(), RS, HiSR) 0
860 .addImm(S) 0
861 .addImm(32-S); 0
862 } ----
863 } else if (S == 32) { 0
864 BuildMI(B, MI, DL, TII->get(TargetOpcode::COPY), (Left ? HiR : LoR)) 0
865 .addReg(Op1.getReg(), RS & ~RegState::Kill, (Left ? LoSR : HiSR)); 0
866 if (!Signed) 0
867 BuildMI(B, MI, DL, TII->get(A2_tfrsi), (Left ? LoR : HiR)) 0
868 .addImm(0); 0
869 else // Must be right shift. ----
870 BuildMI(B, MI, DL, TII->get(S2_asr_i_r), HiR) 0
871 .addReg(Op1.getReg(), RS, HiSR) 0
872 .addImm(31); 0
873 } else if (S < 64) { 0
874 S -= 32; 0
875 if (S == 16 && Left) 0
876 BuildMI(B, MI, DL, TII->get(A2_aslh), HiR) 0
877 .addReg(Op1.getReg(), RS & ~RegState::Kill, LoSR); 0
878 else if (S == 16 && Signed) 0
879 BuildMI(B, MI, DL, TII->get(A2_asrh), LoR) 0
880 .addReg(Op1.getReg(), RS & ~RegState::Kill, HiSR); 0
881 else ----
882 BuildMI(B, MI, DL, TII->get(ShiftOpc), (Left ? HiR : LoR)) 0
883 .addReg(Op1.getReg(), RS & ~RegState::Kill, (Left ? LoSR : HiSR)) 0
884 .addImm(S); 0
885 ----
886 if (Signed) 0
887 BuildMI(B, MI, DL, TII->get(S2_asr_i_r), HiR) 0
888 .addReg(Op1.getReg(), RS, HiSR) 0
889 .addImm(31); 0
890 else ----
891 BuildMI(B, MI, DL, TII->get(A2_tfrsi), (Left ? LoR : HiR)) 0
892 .addImm(0); 0
893 } ----
894 } 0

898 using namespace Hexagon; ----
899 ----

901 MachineOperand &Op1 = MI->getOperand(1); 0
902 MachineOperand &Op2 = MI->getOperand(2); 0

904 assert(Op0.isReg() && Op1.isReg() && Op2.isReg() && Op3.isImm()); 0

908 ----

910 assert(F != PairMap.end()); 0

912 unsigned LoR = P.first; 0

919 const TargetRegisterClass *IntRC = &IntRegsRegClass; 0
920 ----
921 unsigned LoSR = isub_lo; 0
922 unsigned HiSR = isub_hi; 0

925 // means: Op0 = or (Op1, asl(Op2, Op3)) ----
926 ----
927 // Expansion of ----
928 // DR = or (R1, asl(R2, #s)) ----
929 // ----
930 // LoR = or (R1.lo, asl(R2.lo, #s)) ----
931 // Tmp1 = extractu R2.lo, #s, #32-s ----
932 // Tmp2 = or R1.hi, Tmp1 ----
933 // HiR = or (Tmp2, asl(R2.hi, #s)) ----
934 ----
935 if (S == 0) { 0
936 // DR = or (R1, asl(R2, #0)) ----
937 // -> or (R1, R2) ----
938 // i.e. LoR = or R1.lo, R2.lo ----
939 // HiR = or R1.hi, R2.hi ----
940 BuildMI(B, MI, DL, TII->get(A2_or), LoR) 0
941 .addReg(Op1.getReg(), RS1 & ~RegState::Kill, LoSR) 0
942 .addReg(Op2.getReg(), RS2 & ~RegState::Kill, LoSR); 0
943 BuildMI(B, MI, DL, TII->get(A2_or), HiR) 0
944 .addReg(Op1.getReg(), RS1, HiSR) 0
945 .addReg(Op2.getReg(), RS2, HiSR); 0
946 } else if (S < 32) { 0
947 BuildMI(B, MI, DL, TII->get(S2_asl_i_r_or), LoR) 0
948 .addReg(Op1.getReg(), RS1 & ~RegState::Kill, LoSR) 0
949 .addReg(Op2.getReg(), RS2 & ~RegState::Kill, LoSR) 0
950 .addImm(S); 0
951 Register TmpR1 = MRI->createVirtualRegister(IntRC); 0
952 BuildMI(B, MI, DL, TII->get(S2_extractu), TmpR1) 0
953 .addReg(Op2.getReg(), RS2 & ~RegState::Kill, LoSR) 0
954 .addImm(S) 0
955 .addImm(32-S); 0
956 Register TmpR2 = MRI->createVirtualRegister(IntRC); 0
957 BuildMI(B, MI, DL, TII->get(A2_or), TmpR2) 0
958 .addReg(Op1.getReg(), RS1, HiSR) 0
959 .addReg(TmpR1); 0
960 BuildMI(B, MI, DL, TII->get(S2_asl_i_r_or), HiR) 0
961 .addReg(TmpR2) 0
962 .addReg(Op2.getReg(), RS2, HiSR) 0
963 .addImm(S); 0
964 } else if (S == 32) { 0
965 // DR = or (R1, asl(R2, #32)) ----
966 // -> or R1, R2.lo ----
967 // LoR = R1.lo ----
968 // HiR = or R1.hi, R2.lo ----
969 BuildMI(B, MI, DL, TII->get(TargetOpcode::COPY), LoR) 0
970 .addReg(Op1.getReg(), RS1 & ~RegState::Kill, LoSR); 0
971 BuildMI(B, MI, DL, TII->get(A2_or), HiR) 0
972 .addReg(Op1.getReg(), RS1, HiSR) 0
973 .addReg(Op2.getReg(), RS2, LoSR); 0
974 } else if (S < 64) { 0
975 // DR = or (R1, asl(R2, #s)) ----
976 // ----
977 // LoR = R1:lo ----
978 // HiR = or (R1:hi, asl(R2:lo, #s-32)) ----
979 S -= 32; 0
980 BuildMI(B, MI, DL, TII->get(TargetOpcode::COPY), LoR) 0
981 .addReg(Op1.getReg(), RS1 & ~RegState::Kill, LoSR); 0
982 BuildMI(B, MI, DL, TII->get(S2_asl_i_r_or), HiR) 0
983 .addReg(Op1.getReg(), RS1, HiSR) 0
984 .addReg(Op2.getReg(), RS2, LoSR) 0
985 .addImm(S); 0
986 } ----
