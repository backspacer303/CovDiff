9 /// \file ----
10 /// Replaces repeated sequences of instructions with function calls. ----
11 /// ----
12 /// This works by placing every instruction from every basic block in a ----
13 /// suffix tree, and repeatedly querying that tree for repeated sequences of ----
14 /// instructions. If a sequence of instructions appears often, then it ought ----
15 /// to be beneficial to pull out into a function. ----
16 /// ----
17 /// The MachineOutliner communicates with a given target using hooks defined in ----
18 /// TargetInstrInfo.h. The target supplies the outliner with information on how ----
19 /// a specific sequence of instructions should be outlined. This information ----
20 /// is used to deduce the number of instructions necessary to ----
21 /// ----
22 /// * Create an outlined function ----
23 /// * Call that outlined function ----
24 /// ----
25 /// Targets must implement ----
26 /// * getOutliningCandidateInfo ----
27 /// * buildOutlinedFrame ----
28 /// * insertOutlinedCall ----
29 /// * isFunctionSafeToOutlineFrom ----
30 /// ----
31 /// in order to make use of the MachineOutliner. ----
32 /// ----
33 /// This was originally presented at the 2016 LLVM Developers' Meeting in the ----
34 /// talk "Reducing Code Size Using Outlining". For a high-level overview of ----
35 /// how this pass works, the talk is available on YouTube at ----
36 /// ----
37 /// https://www.youtube.com/watch?v=yorld-WSOeU ----
38 /// ----
39 /// The slides for the talk are available at ----
40 /// ----
41 /// http://www.llvm.org/devmtg/2016-11/Slides/Paquette-Outliner.pdf ----
42 /// ----
43 /// The talk provides an overview of how the outliner finds candidates and ----
44 /// ultimately outlines them. It describes how the main data structure for this ----
45 /// pass, the suffix tree, is queried and purged for candidates. It also gives ----

616 // * End before the other starts ----
617 // * Start after the other ends ----
618 unsigned EndIdx = StartIdx + StringLen - 1; 0
619 auto FirstOverlap = find_if( 0
620 CandidatesForRepeatedSeq, [StartIdx, EndIdx](const Candidate &C) { 0
621 return EndIdx >= C.getStartIdx() && StartIdx <= C.getEndIdx(); 0
622 }); ----
623 if (FirstOverlap != CandidatesForRepeatedSeq.end()) { 0
624 #ifndef NDEBUG ----
625 ++NumDiscarded; 0
626 LLVM_DEBUG(dbgs() << " .. DISCARD candidate @ [" << StartIdx 0
627 << ", " << EndIdx << "]; overlaps with candidate @ [" ----
628 << FirstOverlap->getStartIdx() << ", " ----
629 << FirstOverlap->getEndIdx() << "]\n"); ----
630 #endif ----
631 continue; 0
632 } ----
633 // It doesn't overlap with anything, so we can outline it. ----
634 // Each sequence is over [StartIt, EndIt]. ----
635 // Save the candidate and its location. ----
636 #ifndef NDEBUG ----
637 ++NumKept; 0
638 #endif ----
639 MachineBasicBlock::iterator StartIt = Mapper.InstrList[StartIdx]; 0
640 MachineBasicBlock::iterator EndIt = Mapper.InstrList[EndIdx]; 0
641 MachineBasicBlock *MBB = StartIt->getParent(); 0
642 CandidatesForRepeatedSeq.emplace_back(StartIdx, StringLen, StartIt, EndIt, 0
643 MBB, FunctionList.size(), 0
644 Mapper.MBBFlagsMap[MBB]); ----
645 } ----
646 #ifndef NDEBUG ----
647 LLVM_DEBUG(dbgs() << " Candidates discarded: " << NumDiscarded 0
648 << "\n"); ----
649 LLVM_DEBUG(dbgs() << " Candidates kept: " << NumKept << "\n\n"); 0
650 #endif ----
651 ----
652 // We've found something we might want to outline. ----
653 // Create an OutlinedFunction to store it and check if it'd be beneficial ----
654 // to outline. ----
655 if (CandidatesForRepeatedSeq.size() < 2) 0
656 continue; 0
657 ----
658 // Arbitrarily choose a TII from the first candidate. ----
659 // FIXME: Should getOutliningCandidateInfo move to TargetMachine? ----
660 const TargetInstrInfo *TII = ----
661 CandidatesForRepeatedSeq[0].getMF()->getSubtarget().getInstrInfo(); 0
662 ----
663 std::optional OF = ----
664 TII->getOutliningCandidateInfo(CandidatesForRepeatedSeq); 0
665 ----
666 // If we deleted too many candidates, then there's nothing worth outlining. ----
667 // FIXME: This should take target-specified instruction sizes into account. ----
668 if (!OF || OF->Candidates.size() < 2) 0
669 continue; 0
670 ----
671 // Is it better to outline this candidate than not? ----
672 if (OF->getBenefit() < OutlinerBenefitThreshold) { 0
673 emitNotOutliningCheaperRemark(StringLen, CandidatesForRepeatedSeq, *OF); 0
674 continue; 0
675 } ----
676 ----
677 FunctionList.push_back(*OF); 0
678 } 0

823 bool MachineOutliner::outline(Module &M, 0
824 std::vector &FunctionList, ----
825 InstructionMapper &Mapper, ----
826 unsigned &OutlinedFunctionNum) { ----

828 LLVM_DEBUG(dbgs() << "NUMBER OF POTENTIAL FUNCTIONS: " << FunctionList.size() 0
829 << "\n"); ----
830 bool OutlinedSomething = false; 0
831 ----
832 // Sort by benefit. The most beneficial functions should be outlined first. ----
833 stable_sort(FunctionList, 0
834 [](const OutlinedFunction &LHS, const OutlinedFunction &RHS) { 0
835 return LHS.getBenefit() > RHS.getBenefit(); 0
836 }); ----
837 ----
838 // Walk over each function, outlining them as we go along. Functions are ----
839 // outlined greedily, based off the sort above. ----
840 auto *UnsignedVecBegin = Mapper.UnsignedVec.begin(); 0
841 LLVM_DEBUG(dbgs() << "WALKING FUNCTION LIST\n"); 0
842 for (OutlinedFunction &OF : FunctionList) { 0
843 #ifndef NDEBUG ----
844 auto NumCandidatesBefore = OF.Candidates.size(); 0
845 #endif ----
846 // If we outlined something that overlapped with a candidate in a previous ----
847 // step, then we can't outline from it. ----
848 erase_if(OF.Candidates, [&UnsignedVecBegin](Candidate &C) { 0
849 return std::any_of(UnsignedVecBegin + C.getStartIdx(), 0
850 UnsignedVecBegin + C.getEndIdx() + 1, [](unsigned I) { 0
851 return I == static_cast(-1); 0
852 }); 0
853 }); ----
854 ----
855 #ifndef NDEBUG ----
856 auto NumCandidatesAfter = OF.Candidates.size(); 0
857 LLVM_DEBUG(dbgs() << "PRUNED: " << NumCandidatesBefore - NumCandidatesAfter 0
858 << "/" << NumCandidatesBefore << " candidates\n"); ----
859 #endif ----
860 ----
861 // If we made it unbeneficial to outline this function, skip it. ----
862 if (OF.getBenefit() < OutlinerBenefitThreshold) { 0
863 LLVM_DEBUG(dbgs() << "SKIP: Expected benefit (" << OF.getBenefit() 0
864 << " B) < threshold (" << OutlinerBenefitThreshold ----
865 << " B)\n"); ----
866 continue; 0
867 } ----
868 ----
869 LLVM_DEBUG(dbgs() << "OUTLINE: Expected benefit (" << OF.getBenefit() 0
870 << " B) > threshold (" << OutlinerBenefitThreshold ----
871 << " B)\n"); ----
872 ----
873 // It's beneficial. Create the function and outline its sequence's ----
874 // occurrences. ----
875 OF.MF = createOutlinedFunction(M, OF, Mapper, OutlinedFunctionNum); 0
876 emitOutlinedFunctionRemark(OF); 0
877 FunctionsCreated++; 0
878 OutlinedFunctionNum++; // Created a function, move to the next name. 0
879 MachineFunction *MF = OF.MF; 0
880 const TargetSubtargetInfo &STI = MF->getSubtarget(); 0
881 const TargetInstrInfo &TII = *STI.getInstrInfo(); 0
882 ----
883 // Replace occurrences of the sequence with calls to the new function. ----
884 LLVM_DEBUG(dbgs() << "CREATE OUTLINED CALLS\n"); 0
885 for (Candidate &C : OF.Candidates) { 0
886 MachineBasicBlock &MBB = *C.getMBB(); 0
887 MachineBasicBlock::iterator StartIt = C.front(); 0
888 MachineBasicBlock::iterator EndIt = C.back(); 0
889 ----
890 // Insert the call. ----
891 auto CallInst = TII.insertOutlinedCall(M, MBB, StartIt, *MF, C); 0
892 // Insert the call. ----
893 #ifndef NDEBUG ----
894 auto MBBBeingOutlinedFromName = ----
895 MBB.getName().empty() ? "" : MBB.getName().str(); 0
896 auto MFBeingOutlinedFromName = MBB.getParent()->getName().empty() 0
897 ? "" ----
898 : MBB.getParent()->getName().str(); 0
899 LLVM_DEBUG(dbgs() << " CALL: " << MF->getName() << " in " 0
900 << MFBeingOutlinedFromName << ":" ----
901 << MBBBeingOutlinedFromName << "\n"); ----
902 LLVM_DEBUG(dbgs() << " .. " << *CallInst); 0
903 #endif ----
904 ----
905 // If the caller tracks liveness, then we need to make sure that ----
906 // anything we outline doesn't break liveness assumptions. The outlined ----
907 // functions themselves currently don't track liveness, but we should ----
908 // make sure that the ranges we yank things out of aren't wrong. ----
909 if (MBB.getParent()->getProperties().hasProperty( 0
910 MachineFunctionProperties::Property::TracksLiveness)) { ----
911 // The following code is to add implicit def operands to the call ----
912 // instruction. It also updates call site information for moved ----
913 // code. ----
914 SmallSet UseRegs, DefRegs; 0
915 // Copy over the defs in the outlined range. ----
916 // First inst in outlined range <-- Anything that's defined in this ----
917 // ... .. range has to be added as an ----
918 // implicit Last inst in outlined range <-- def to the call ----
919 // instruction. Also remove call site information for outlined block ----
920 // of code. The exposed uses need to be copied in the outlined range. ----
921 for (MachineBasicBlock::reverse_iterator 0
922 Iter = EndIt.getReverse(), 0
923 Last = std::next(CallInst.getReverse()); 0
924 Iter != Last; Iter++) { 0
925 MachineInstr *MI = &*Iter; 0
926 SmallSet InstrUseRegs; 0
927 for (MachineOperand &MOP : MI->operands()) { 0
928 // Skip over anything that isn't a register. ----
929 if (!MOP.isReg()) 0
930 continue; 0
931 ----
932 if (MOP.isDef()) { 0
933 // Introduce DefRegs set to skip the redundant register. ----
934 DefRegs.insert(MOP.getReg()); 0
935 if (UseRegs.count(MOP.getReg()) && 0
936 !InstrUseRegs.count(MOP.getReg())) 0
937 // Since the regiester is modeled as defined, ----
938 // it is not necessary to be put in use register set. ----
939 UseRegs.erase(MOP.getReg()); 0
940 } else if (!MOP.isUndef()) { 0
941 // Any register which is not undefined should ----
942 // be put in the use register set. ----
943 UseRegs.insert(MOP.getReg()); 0
944 InstrUseRegs.insert(MOP.getReg()); 0
945 } ----
946 } ----
947 if (MI->isCandidateForCallSiteEntry()) 0
948 MI->getMF()->eraseCallSiteInfo(MI); 0
949 } 0
950 ----
951 for (const Register &I : DefRegs) 0
952 // If it's a def, add it to the call instruction. ----
953 CallInst->addOperand( 0
954 MachineOperand::CreateReg(I, true, /* isDef = true */ 0
955 true /* isImp = true */)); 0
956 ----
957 for (const Register &I : UseRegs) 0
958 // If it's a exposed use, add it to the call instruction. ----
959 CallInst->addOperand( 0
960 MachineOperand::CreateReg(I, false, /* isDef = false */ 0
961 true /* isImp = true */)); 0
962 } 0
963 ----
964 // Erase from the point after where the call was inserted up to, and ----
965 // including, the final instruction in the sequence. ----
966 // Erase needs one past the end, so we need std::next there too. ----
967 MBB.erase(std::next(StartIt), std::next(EndIt)); 0
968 ----
969 // Keep track of what we removed by marking them all as -1. ----
970 for (unsigned &I : make_range(UnsignedVecBegin + C.getStartIdx(), 0
971 UnsignedVecBegin + C.getEndIdx() + 1)) 0
972 I = static_cast(-1); 0
973 OutlinedSomething = true; 0
974 ----
975 // Statistics. ----
976 NumOutlined++; 0
977 } 0
978 } ----
979 ----
980 LLVM_DEBUG(dbgs() << "OutlinedSomething = " << OutlinedSomething << "\n";); 0
981 return OutlinedSomething; 0

1157 // -enable-machine-outliner, the pass will run on all functions in the module. ----
1158 // Otherwise, if the target supports default outlining, it will run on all ----
1159 // functions deemed by the target to be worth outlining from by default. Tell ----
1160 // the user how the outliner is running. ----
1161 LLVM_DEBUG({ 0

1164 dbgs() << "all functions"; ----
1165 else ----
1166 dbgs() << "target-default functions"; ----
1167 dbgs() << "\n"; ----
1168 }); ----
1169 ----

1171 // it here. ----
1172 OutlineFromLinkOnceODRs = EnableLinkOnceODROutlining; 0

1174 ----
1175 // Prepare instruction mappings for the suffix tree. ----
1176 populateMapper(Mapper, M, MMI); 0
1177 std::vector FunctionList; 0
1178 ----
1179 // Find all of the outlining candidates. ----
1180 findCandidates(Mapper, FunctionList); 0
1181 ----
1182 // If we've requested size remarks, then collect the MI counts of every ----
1183 // function before outlining, and the MI counts after outlining. ----
1184 // FIXME: This shouldn't be in the outliner at all; it should ultimately be ----
1185 // the pass manager's responsibility. ----
1186 // This could pretty easily be placed in outline instead, but because we ----
1187 // really ultimately *don't* want this here, it's done like this for now ----
1188 // instead. ----
1189 ----
1190 // Check if we want size remarks. ----
1191 bool ShouldEmitSizeRemarks = M.shouldEmitInstrCountChangedRemark(); 0
